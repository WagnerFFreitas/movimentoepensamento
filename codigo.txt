        body: JSON.stringify({ email, senha })
      });

      if (response.ok) {
        const data = await response.json();
        localStorage.setItem('token', data.token);
        window.location.href = 'dashboard.html';
      } else {
        // Tratar erros de login (e.g., exibir mensagem de erro ao usuário)
        console.error('Erro de login:', await response.text()); 
      }
    } catch (error) {
      console.error('Erro de login:', error);
      // Tratar erros de rede (e.g., exibir mensagem de erro ao usuário)
    }
  },

  verificarAuth() {
    return !!localStorage.getItem('token');
  },

  logout() {
    localStorage.removeItem('token');
    window.location.href = 'login.html';
  }
};

// Carrossel de Testemunhos
document.addEventListener('DOMContentLoaded', function() {
  const slides = document.querySelectorAll('.testimonial-container > div');
  const indicators = document.querySelectorAll('.testimonial-indicators .indicator');
  let currentSlide = 0;

  function showSlide(index) {
    slides.forEach(slide => slide.classList.remove('active'));
    indicators.forEach(indicator => indicator.classList.remove('active'));

    slides[index].classList.add('active');
    indicators[index].classList.add('active');
  }

  function nextSlide() {
    currentSlide = (currentSlide + 1) % slides.length;
    showSlide(currentSlide);
  }

  setInterval(nextSlide, 5000); // Intervalo de 5 segundos

  indicators.forEach((indicator, index) => {
    indicator.addEventListener('click', () => {
      currentSlide = index;
      showSlide(currentSlide);
    });
  });

  showSlide(currentSlide);
});
