<template>
  <div class="atendimento-detalhes">
    <el-card>
      <template #header>
        <div class="card-header">
          <h2>Detalhes do Atendimento</h2>
        </div>
      </template>

      <el-form :model="atendimento" label-width="120px">
        <el-row :gutter="20">
          <el-col :span="12">
            <el-form-item label="Paciente">
              <el-input v-model="atendimento.paciente.nome" disabled />
            </el-form-item>
          </el-col>
          
          <el-col :span="12">
            <el-form-item label="Data/Hora">
              <el-input 
                :value="formatarDataHora(atendimento.dataHora)" 
                disabled 
              />
            </el-form-item>
          </el-col>
        </el-row>

        <el-form-item label="Evolução">
          <el-input 
            type="textarea" 
            v-model="atendimento.evolucao" 
            :rows="4"
          />
        </el-form-item>

        <el-form-item label="Conduta">
          <el-input 
            type="textarea" 
            v-model="atendimento.conduta" 
            :rows="4"
          />
        </el-form-item>

        <el-form-item>
          <el-button 
            type="primary" 
            @click="salvarAtendimento"
            :loading="salvando"
          >
            Salvar Atendimento
          </el-button>
        </el-form-item>
      </el-form>
    </el-card>
  </div>
</template>

<script>
import { ref } from 'vue';
import AtendimentoService from '@/services/AtendimentoService';

export default {
  name: 'AtendimentoDetalhes',
  
  props: {
    atendimentoId: {
      type: String,
      required: true
    }
  },

  setup(props) {
    const atendimento = ref({
      paciente: {},
      dataHora: null,
      evolucao: '',
      conduta: ''
    });
    const salvando = ref(false);

    const buscarAtendimento = async () => {
      try {
        const resposta = await AtendimentoService.buscarAtendimento(props.atendimentoId);
        atendimento.value = resposta.data;
      } catch (erro) {
        ElMessage.error('Erro ao carregar atendimento');
      }
    };

    const salvarAtendimento = async () => {
      salvando.value = true;
      try {
        await AtendimentoService.salvarAtendimento(props.atendimentoId, {
          evolucao: atendimento.value.evolucao,
          conduta: atendimento.value.conduta
        });
        ElMessage.success('Atendimento salvo com sucesso');
      } catch (erro) {
        ElMessage.error('Erro ao salvar atendimento');
      } finally {
        salvando.value = false;
      }
    };

    const formatarDataHora = (dataHora) => {
      return new Date(dataHora).toLocaleString();
    };

    // Carregar dados iniciais
    buscarAtendimento();

    return {
      atendimento,
      salvando,
      salvarAtendimento,
      formatarDataHora
    };
  }
};
</script>

// tests/unit/AtendimentoService.spec.js
import { describe, it, expect, beforeEach, vi } from 'vitest';
import AtendimentoService from '@/services/AtendimentoService';
import ApiClient from '@/services/ApiClient';

vi.mock('@/services/ApiClient');

describe('AtendimentoService', () => {
  beforeEach(() => {
    vi.clearAllMocks();
  });

  it('deve buscar atendimento corretamente', async () => {
    const mockAtendimento = {
      id: '1',
      paciente: { nome: 'João' },
      dataHora: '2024-01-20T10:00:00'
    };

    ApiClient.get.mockResolvedValue({ data: mockAtendimento });

    const resultado = await AtendimentoService.buscarAtendimento('1');
    expect(resultado.data).toEqual(mockAtendimento);
    expect(ApiClient.get).toHaveBeenCalledWith('/atendimentos/1');
  });

  it('deve salvar atendimento corretamente', async () => {
    const dadosAtendimento = {
      evolucao: 'Evolução do paciente',
      conduta: 'Conduta aplicada'
    };

    ApiClient.put.mockResolvedValue({ data: { sucesso: true } });

    const resultado = await AtendimentoService.salvarAtendimento('1', dadosAtendimento);
    expect(resultado.data.sucesso).toBe(true);
    expect(ApiClient.put).toHaveBeenCalledWith('/atendimentos/1', dadosAtendimento);
  });
});
